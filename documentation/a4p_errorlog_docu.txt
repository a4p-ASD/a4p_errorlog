	/etc/httpd/conf.d/
	
		error_logs.conf
		
			
<Directory "/var/www/vhosts/_a4p_apache_logs">

	Options Indexes MultiViews FollowSymLinks
	AllowOverride None
	Order allow,deny
	Allow from all

# Virtual hosts oxid-develop.apps4print.com

<VirtualHost *>
	ServerName				oxid-develop.apps4print.com
	ServerAlias				oxid-develop.apps4print.com
	DocumentRoot			/var/www/vhosts/oxid-develop.apps4print.com
	ErrorLog				/var/www/vhosts/_a4p_apache_logs/oxid-develop/error_log
